# Clone Repo:

- desktop (open git bash)
- git clone URL (repository)
- cd file name
- code . (VS CODE)

# Front-end Config:

- cd view
- npm i / npm install
- npm run dev

# Back-end Config:

- composer install
- cp .env.example .env
- php artisan key:generate
- php artisan migrate --seed
- php artisan serve
